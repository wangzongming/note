事件冒泡:当一个元素接收到时间的时候 会把它接收的事件传播给父级

div3.onclick=function(ev){
    ev=ev||window.event;
    ev.cancelbubble=true;  //阻止冒泡
    alert("不会向上传递事件了")
}
